<h2>Controls</h2>

<table>
<tr>
    <td>F2</td>
    <td>Toggle bot</td>
</tr>
<tr>
    <td>F4</td>
    <td>Toggle freeze</td>
</tr>
<tr>
    <td>W / ←</td>
    <td>Move player 1 up</td>
</tr>
<tr>
    <td>Y / →</td>
    <td>Move player 1 down</td>
</tr>
<tr>
    <td>A / ↑</td>
    <td>Move player 2 u</td>
</tr>
<tr>
    <td>S / ↓</td>
    <td>Move player 2 down</td>
</tr>
</table>
